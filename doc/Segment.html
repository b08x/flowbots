def add_word(word_data)
  word = Word.create(word_data.merge(segment: self))
  words.push(word)
  save

def add_words(new_words)
  new_words.each do |word_data|
    word = Word.create(word_data.merge(segment: self))
    words.push(word)
  end
  save
end
