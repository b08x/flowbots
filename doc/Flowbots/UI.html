def response(response)
  ui.space
  ui.h1 "UI: Text Line Animation"
  ui.space

  response.each_line do |line|
    input = line.chomp
    unless input.nil?
      input.each_char do |char|
        print "\e[34m#{char}\e[0m"
        ui.message char
        sleep 0.02
      end
      puts "\n"
    end
    res = line.chomp
    unless res.nil?
      puts "\e[32m#{res.strip.chomp}\e[0m"
      puts "\n"
    end
    cap = line.chomp
    puts "\e[33m#{cap.strip.chomp}\e[0m\n" unless cap.nil?
  end
  puts "\n"
  sleep 3
  ui.space
end

def say(type, statement)
  prompt
  type = :ok if type.nil?
  case type
  when :ok
    @prompt.ok(statement)
    logger.info statement
  when :warn
    @prompt.warn(statement)
    logger.warn statement
  when :error
    @prompt.error(statement)
    logger.fatal statement
  else
    PASTEL.say(statement)
  end
end
