def execute
  @logger.info("Starting execution of #{self.class.task_name}")
  result = nil
  begin
    result = yield if block_given?
    @logger.info("Successfully completed execution of #{self.class.task_name}")
  rescue StandardError => e
    handle_error(e)
  end
  result
end
